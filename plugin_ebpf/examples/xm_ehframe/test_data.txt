void generateSortedArray(int arr[], int n) {
    int i;
    int current = 1;  // 初始值
    for (i = 0; i < n; i++) {
        arr[i] = current;
        current += rand() % 5 + 1;  // 随机增加 1 到 5 的值
    }
}
